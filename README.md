# Setup zsh
### zshrc
Copy .zshrc to $HOME

    cp .zshrc $HOME/

### Theme
Copy custom agnoster theme to $HOME/.oh-my-zsh/custom/themes/

    cp agnoster-nanof.zsh-theme $HOME/.oh-my-zsh/custom/themes/


# Setup nvim
Copy init.vim to $HOME/.config/nvim/init.vim

    cp init.vim $HOME/.config/nvim/
